Sub ReverterBinarioParaOriginal() Dim fso As Object Dim pastaOrigem As String, pastaDestino As String Dim arquivo As Object Dim fs As Object Dim binario As String Dim bytes() As Byte Dim i As Long ' Definir as pastas de origem e destino pastaOrigem = "C:\fs1\FSCapture73-20250213T110033Z-001\FSCapture73\FSCapture73\Temp\" pastaDestino = "C:\fs1\FSCapture73-20250213T110033Z-001\FSCapture73\FSCapture73\" ' Criar objeto FileSystemObject Set fso = CreateObject("Scripting.FileSystemObject") ' Percorrer arquivos na pasta Temp For Each arquivo In fso.GetFolder(pastaOrigem).Files ' Permitir que o sistema processe eventos pendentes DoEvents ' Ler o conte√∫do do arquivo de texto binario = "" Set fs = fso.OpenTextFile(arquivo.Path, 1) ' Modo leitura binario = fs.ReadAll fs.Close Set fs = Nothing ' Converter string hexadecimal de volta para bytes ReDim bytes((Len(binario) \ 2) - 1) For i = 0 To UBound(bytes) bytes(i) = CByte("&H" & Mid(binario, (i * 2) + 1, 2)) Next i ' Salvar os bytes no arquivo original Open pastaDestino & fso.GetBaseName(arquivo.Name) For Binary As #1 Put #1, , bytes Close #1 Next arquivo MsgBox "Arquivos restaurados com sucesso!", vbInformation End Sub
